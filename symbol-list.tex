\documentclass{article}
\usepackage{setspace}
\newfont{\musFont}{musix13}

\NewDocumentCommand{\symbolInFont}{ m m }{{#1\symbol{#2}}}

\newcounter{symbol}
\NewDocumentCommand{\symbolList}{ m m }{%
    \begin{doublespace}
        \setcounter{symbol}{1}
        \noindent
        \loop
            \makebox[4em]{\thesymbol\quad\symbolInFont{#1}{\value{symbol}}}
            \stepcounter{symbol}
            \ifnum\value{symbol} < #2
         \repeat
    \end{doublespace}
}
\begin{document}

\symbolList{\musFont}{126}

\end{document}
